### 3.2 е№¶иЎҢй…ҚзҪ®

**д»Јз ҒдҪҚзҪ®**: `xllm/core/framework/parallel_state/`

```cpp
// й…ҚзҪ®зӨәдҫӢ
runtime::Options options;
options.tp_size = 8;           // еј йҮҸе№¶иЎҢеәҰ
options.pp_size = 1;          // жөҒж°ҙзәҝе№¶иЎҢеәҰ  
options.dp_size = 4;          // ж•°жҚ®е№¶иЎҢеәҰ
options.cp_size = 2;          // дёҠдёӢж–Үе№¶иЎҢеәҰ
options.ep_size = 8;          // дё“е®¶е№¶иЎҢеәҰ

// жҖ»Workerж•° = dp_size Г— tp_size Г— pp_size Г— cp_size
// дҫӢеҰӮ: 4 Г— 8 Г— 1 Г— 2 = 64 дёӘWorkerиҝӣзЁӢ
```

### 7.2 д»Јз ҒдҪҚзҪ®йӘҢиҜҒ

```bash
# йӘҢиҜҒжһ¶жһ„дёҺд»Јз ҒдҪҚзҪ®дёҖиҮҙжҖ§
#!/bin/bash

EXPECTED_LOCATIONS=(
    "Scheduler:core/scheduler"
    "Worker:core/runtime"
    "Batch:core/framework/batch"
    "Request:core/framework/request"
    "KVCache:core/framework/kv_cache"
    "BlockManager:core/framework/block"
    "CausalLM:core/framework/model"
    "ParallelState:core/framework/parallel_state"
)

for entry in "${EXPECTED_LOCATIONS[@]}"; do
    IFS=':' read -r class path <<< "$entry"
    if grep -rq "class $class" "xllm/$path"; then
        echo "вң“ $class found in xllm/$path"
    else
        echo "вң— $class NOT found in xllm/$path"
    fi
done
```
